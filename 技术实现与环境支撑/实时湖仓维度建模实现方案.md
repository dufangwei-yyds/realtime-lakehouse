# 实时湖仓维度建模实现方案

> 以下为对应 SQL 脚本的完整内容，已原样搬运，便于在仅支持 Markdown 的设备中查看。

```sql
/*
=============================================================
  服装业务实时湖仓维度建模
  DIM层(Dimension, 维度层): 基于维度建模设计的公共维度表层, 为后续的明细层(DWD)和
  汇总层(DWS)提供统一的描述性上下文
  DIM提供"相关方是谁/在什么环境下发生"(静态/缓慢变化属性)
  数据构成:
    高基数维度: 数据量庞大且频繁更新的实体, 如用户资料表(千万/亿级)、商品信息表
    低基数维度: 稳定的配置表或枚举值, 如区域编码、日期维表、支付方式定义
  在实时流处理中的角色:
    流表关联(Stream-Table Join): 在实时计算(如Flink任务)中, ODS层的事实流会通过关联DIM层来"打宽"数据,
    将编码(ID)转换为可读的标签或名称
    数据一致性: 确保整个企业内部对同一维度的定义是统一的(如所有报表中的"城市代码"含义一致), 避免出现"烟囱式"开发
  核心: 维度退化(物理打平维度宽表)、缓慢变化维(append、history、latest)

  业务场景：服装SKU的基础信息、价格体系、季节属性三大维度的实时建模
  架构设计：MySQL业务库 → Flink CDC → Paimon湖仓
  维度分层：
    1. Append表：存储原始变更记录，作为SCD2的基础
    2. SCD2历史表：存储维度全量历史版本，支持时间范围回溯
    3. Latest镜像表：存储维度最新状态，用于高效关联查询
    4. 维度退化宽表：将多维度表物理打平，减少查询时的JOIN开销
  前置条件：
    1. Flink已部署并加载mysql-cdc-connector、paimon-connector
    2. HDFS集群正常运行，Paimon Warehouse路径可访问
    3. MySQL开启binlog（ROW模式），且Flink有MySQL读写权限
=============================================================
*/

'fields.end_time.aggregate-function' = 'last_value', -- 取最新结束时间
'fields.end_time.ignore-retract' = 'true', -- 忽略回撤消息
'changelog-producer' = 'lookup' -- 变更日志生成方式：基于查询（支持版本回溯）

CREATE TABLE dim_price_latest (
sku_id INT PRIMARY KEY NOT ENFORCED,

CREATE TABLE dim_attr_latest (
sku_id INT PRIMARY KEY NOT ENFORCED,

'connector' = 'paimon',

CREATE TABLE dim_sku_wide (
sku_id INT PRIMARY KEY NOT ENFORCED, -- SKU主键
sku_name STRING, -- 基础信息：商品名称
color STRING, -- 基础信息：颜色
tag_price DECIMAL(10, 2), -- 价格信息：吊牌价
season STRING, -- 属性信息：季节
update_time TIMESTAMP(3), -- 宽表最新更新时间（取三大维度最新时间）
WATERMARK FOR update_time AS update_time - INTERVAL '5' SECOND
) WITH (
'connector' = 'paimon',
'changelog-producer' = 'lookup',
'merge-engine' = 'partial-update', -- 合并引擎：部分更新（仅更新变更字段）
'ignore-delete' = 'true', -- 忽略删除操作（维度表不删除历史）
'sequence.field' = 'update_time' -- 序列字段：基于更新时间保留最新版本
);
-- ==============================================
-- 步骤5：Flink SQL 实现数据写入逻辑（双写+宽表关联）
-- 说明：
-- - 双写逻辑：同时写入Append表、SCD2历史表、Latest镜像表
-- - 宽表关联：基于Latest表关联生成退化宽表
-- ==============================================

-- 5.1 配置流执行参数
SET 'execution.checkpointing.interval' = '10s'; -- 检查点间隔（流任务必配）
SET 'execution.runtime-mode' = 'streaming'; -- 运行模式：流模式（CDC增量处理）
SET 'table.exec.state.ttl' = '24h'; -- 状态过期时间：24小时
SET 'table.exec.sink.upsert-materialize' = 'NONE'; -- 禁用Upsert物化，提升性能

-- 5.2 一流多写：同时维护SCD2历史表和Latest镜像表
BEGIN STATEMENT SET;

-- 5.2.1 基础信息流处理
-- 写入Append表（原始变更记录）
INSERT INTO dim_base_append
SELECT sku_id, sku_name, color, CAST(PROCTIME() AS TIMESTAMP(3)) AS start_time
FROM mysql_goods_base;

-- 基于Append表生成SCD2历史表（LEAD函数计算版本结束时间）
INSERT INTO dim_base_history
SELECT
sku_id,
sku_name,
color,
start_time,
-- LEAD函数：取下一个版本的开始时间作为当前版本的结束时间，无则设为9999-12-31
LEAD(start_time, 1, TIMESTAMP '9999-12-31 23:59:59.999')
OVER (PARTITION BY sku_id ORDER BY start_time) AS end_time
FROM dim_base_append;

-- 写入Latest镜像表（仅保留最新状态）
INSERT INTO dim_base_latest
SELECT sku_id, sku_name, color, updated_at FROM mysql_goods_base;

-- 5.2.2 价格信息流处理
INSERT INTO dim_price_append
SELECT sku_id, tag_price, CAST(PROCTIME() AS TIMESTAMP(3)) AS start_time
FROM mysql_goods_price;

INSERT INTO dim_price_history
SELECT
sku_id,
tag_price,
start_time,
LEAD(start_time, 1, TIMESTAMP '9999-12-31 23:59:59.999')
OVER (PARTITION BY sku_id ORDER BY start_time) AS end_time
FROM dim_price_append;

INSERT INTO dim_price_latest
SELECT sku_id, tag_price, updated_at FROM mysql_goods_price;

-- 5.2.3 季节属性信息流处理
INSERT INTO dim_attr_append
SELECT sku_id, season, CAST(PROCTIME() AS TIMESTAMP(3)) AS start_time
FROM mysql_goods_attr;

INSERT INTO dim_attr_history
SELECT
sku_id,
season,
start_time,
LEAD(start_time, 1, TIMESTAMP '9999-12-31 23:59:59.999')
OVER (PARTITION BY sku_id ORDER BY start_time) AS end_time
FROM dim_attr_append;

INSERT INTO dim_attr_latest
SELECT sku_id, season, updated_at FROM mysql_goods_attr;

END;

-- 5.3 写入维度退化宽表（关联三个Latest表，物理打平）
INSERT INTO dim_sku_wide
SELECT
COALESCE(b.sku_id, p.sku_id, a.sku_id) as sku_id, -- 兼容部分维度缺失的场景
b.sku_name,
b.color,
p.tag_price,
a.season,
-- 取三大维度中最新的更新时间作为宽表更新时间
GREATEST(
IFNULL(b.update_time, TO_TIMESTAMP('1970-01-01 00:00:00.000')), -- 补充毫秒，匹配TIMESTAMP(3)
IFNULL(p.update_time, TO_TIMESTAMP('1970-01-01 00:00:00.000')),
IFNULL(a.update_time, TO_TIMESTAMP('1970-01-01 00:00:00.000'))
) as update_time
FROM dim_base_latest AS b
FULL OUTER JOIN dim_price_latest AS p ON b.sku_id = p.sku_id -- 全外连接，兼容维度缺失
FULL OUTER JOIN dim_attr_latest AS a ON COALESCE(b.sku_id, p.sku_id) = a.sku_id;

-- ==============================================
-- 步骤6：查询验证宽表数据
-- 说明：批模式查询宽表，验证数据是否正确写入
-- ==============================================
SET 'execution.runtime-mode' = 'batch'; -- 切换为批模式查询
SELECT \* FROM dim_sku_wide;
-- 预期结果：101, '针织运动衫', '海军蓝', 299.00, '2025春', [最新更新时间]
```

## ODS 层开发与部署

### 1. MySQL 端创建原始业务表

> **说明**：模拟服装业务的商品实体的 5 张业务表
>
> - `M_PRODUCT` 通过 `m_dim_id` 关联 `M_DIM`
> - SKU = `M_PRODUCT` + `M_PRODUCT_ALIAS`（COLOR + SIZE 组合）
> - 扩展属性 = `M_PRODUCT` → `M_ATTRIBUTESETINSTANCE` → `M_ATTRIBUTE`

```sql
-- ==============================================
-- 切换到 MySQL 客户端执行以下 SQL
-- ==============================================

CREATE DATABASE IF NOT EXISTS erp_db;
USE erp_db;

-- 1.1 M_PRODUCT（商品主表）
CREATE TABLE M_PRODUCT (
    id                VARCHAR(64) PRIMARY KEY,     -- 商品 id
    name              VARCHAR(64) NOT NULL,        -- 款号
    `value`           VARCHAR(128),                -- 品名
    m_dim_id          VARCHAR(64),                 -- 关联 M_DIM
    m_attributesetinstance_id VARCHAR(64),         -- 关联属性实例
    tag_price         DECIMAL(12,2),               -- 吊牌价
    guide_price       DECIMAL(12,2),               -- 零售指导价
    pdt_cost          DECIMAL(12,2),               -- 商品成本
    market_date       DATE,                        -- 上市日
    created_at        DATETIME(3) DEFAULT CURRENT_TIMESTAMP(3),
    updated_at        DATETIME(3) DEFAULT CURRENT_TIMESTAMP(3) ON UPDATE CURRENT_TIMESTAMP(3)
);

-- 1.2 M_DIM（商品维度表）
CREATE TABLE M_DIM (
    id                VARCHAR(64) PRIMARY KEY,     -- 维度 id
    brand_name        VARCHAR(64),                 -- 品牌
    product_year      VARCHAR(16),                 -- 商品年份
    season            VARCHAR(32),                 -- 季节
    wave_band         VARCHAR(64),                 -- 波段
    series            VARCHAR(64),                 -- 系列
    product_type      VARCHAR(64),                 -- 类别
    sex               VARCHAR(16),                 -- 性别
    age_period        VARCHAR(32),                 -- 年龄段
    big_small_type    VARCHAR(32),                 -- 大小件
    is_hat            VARCHAR(8),                  -- 是否连帽
    style_source      VARCHAR(64),                 -- 款式来源
    qdtype_pdt        VARCHAR(16),                 -- 渠道类型
    is_active         VARCHAR(8) DEFAULT 'Y',      -- 可用
    is_gift           VARCHAR(8) DEFAULT 'N',      -- 可作赠品
    is_can_retail     VARCHAR(8) DEFAULT 'Y',      -- 允许零售
    created_at        DATETIME(3),
    updated_at        DATETIME(3)
);

-- 1.3 M_PRODUCT_ALIAS（商品别名表，一对多）
CREATE TABLE M_PRODUCT_ALIAS (
    id                VARCHAR(64) PRIMARY KEY,     -- 商品别名 id
    m_product_id      VARCHAR(64) NOT NULL,        -- 商品 id
    alias_type        VARCHAR(32),                 -- 如：COLOR, SIZE
    alias_value       VARCHAR(64),                 -- 如：BLACK, M
    created_at        DATETIME(3),
    updated_at        DATETIME(3)
);

-- 1.4 M_ATTRIBUTESETINSTANCE（属性集实例）
CREATE TABLE M_ATTRIBUTESETINSTANCE (
    id                VARCHAR(64) PRIMARY KEY,      -- 属性集实例 id
    m_attributeset_id VARCHAR(64),                  -- 属性集 id
    description       VARCHAR(255),                 -- 属性描述
    created_at        DATETIME(3),
    updated_at        DATETIME(3)
);

-- 1.5 M_ATTRIBUTE（属性值明细）
CREATE TABLE M_ATTRIBUTE (
    id                VARCHAR(64) PRIMARY KEY,      -- 属性值明细 id
    m_attributesetinstance_id VARCHAR(64) NOT NULL, -- 属性集实例 id
    attribute_key     VARCHAR(64),                  -- 如：MATERIAL, ORIGIN
    attribute_value   VARCHAR(128),                 -- 如：COTTON, CHINA
    created_at        DATETIME(3),
    updated_at        DATETIME(3)
);
```

```sql
    d.is_gift,
    d.is_can_retail,
    a.color_code,
    a.size_code,
    s.description AS attrset_desc,
    t.material,
    t.origin,
    p.tag_price,
    p.guide_price,
    p.pdt_cost,
    p.market_date,
    GREATEST(
        p.updated_at,
        d.updated_at,
        COALESCE(a.alias_updated_at, TIMESTAMP '1970-01-01 00:00:00'),
        COALESCE(s.updated_at, TIMESTAMP '1970-01-01 00:00:00'),
        COALESCE(t.attr_updated_at, TIMESTAMP '1970-01-01 00:00:00')
    ) AS dim_updated_at
FROM ods.ods_m_product_latest p
LEFT JOIN ods.ods_m_dim_latest d
    ON p.m_dim_id = d.id
LEFT JOIN (
    -- 内联 v_ods_product_alias_pivot
    SELECT
        m_product_id AS sku_id,
        MAX(CASE WHEN alias_type = 'COLOR' THEN alias_value END) AS color_code,
        MAX(CASE WHEN alias_type = 'SIZE' THEN alias_value END) AS size_code,
        MAX(updated_at) AS alias_updated_at
    FROM ods.ods_m_product_alias_latest
    GROUP BY m_product_id
) a ON p.id = a.sku_id
LEFT JOIN ods.ods_m_attributesetinstance_latest s
    ON p.m_attributesetinstance_id = s.id
LEFT JOIN (
    -- 内联 v_ods_attribute_pivot
    SELECT
        m_attributesetinstance_id,
        MAX(CASE WHEN attribute_key = 'MATERIAL' THEN attribute_value END) AS material,
        MAX(CASE WHEN attribute_key = 'ORIGIN' THEN attribute_value END) AS origin,
        MAX(updated_at) AS attr_updated_at
    FROM ods.ods_m_attribute_latest
    GROUP BY m_attributesetinstance_id
) t ON p.m_attributesetinstance_id = t.m_attributesetinstance_id;

-- 6.5 查询验证
SET 'execution.runtime-mode' = 'batch';
SELECT *
FROM dim_sku_wide
ORDER BY sku_id;
```

### 实现说明

- `ods_m_product_latest` 提供 SKU 主体、价格与上市日期
- `ods_m_dim_latest` 提供品牌、年份、季节、波段、系列、品类、渠道等主维度属性
- `ods_m_product_alias_latest` 通过 `alias_type` 透视出颜色和尺码
- `ods_m_attributesetinstance_latest` 提供属性集实例描述
- `ods_m_attribute_latest` 通过 `attribute_key` 透视出材质和产地等扩展属性

### 生产建议

- DWD 层统一使用 `dim_sku_wide` 做商品维度 Lookup Join
- 若后续属性维度继续扩展，可继续在 `v_ods_attribute_pivot` 中补充 `MAX(CASE WHEN ...)` 字段
- 宽表更新时间统一使用多表 `updated_at` 的最大值，便于幂等更新与增量覆盖
